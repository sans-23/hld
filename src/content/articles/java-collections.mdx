export const CodeTabs = ({ children }) => {
  const [activeTab, setActiveTab] = useState(0);
  const tabs = React.Children.toArray(children);
  const activeCode = tabs[activeTab]?.props.code || '';
  const activeLang = tabs[activeTab]?.props.lang || 'text';
  const handleCopy = () => {
    navigator.clipboard.writeText(activeCode);
  };
  return (
    <div className="code-block" style={{ margin: '24px 0' }}>
      <div className="code-block-header" style={{ padding: '6px 16px', display: 'flex', alignItems: 'center', justifyContent: 'space-between' }}>
        <div style={{ display: 'flex', gap: '8px' }}>
          {tabs.map((tab, idx) => (
            <button
              key={idx}
              onClick={() => setActiveTab(idx)}
              style={{
                background: activeTab === idx ? 'rgba(255, 255, 255, 0.12)' : 'none',
                border: 'none',
                color: activeTab === idx ? '#fff' : 'rgba(255, 255, 255, 0.4)',
                padding: '6px 12px',
                borderRadius: '6px',
                fontSize: '0.75rem',
                fontWeight: '600',
                cursor: 'pointer',
                transition: 'all 0.15s ease',
                outline: 'none'
              }}
            >
              {tab.props.label}
            </button>
          ))}
        </div>
        <button 
          className="code-block-copy" 
          onClick={handleCopy}
          style={{
            background: 'none',
            border: '1px solid rgba(255, 255, 255, 0.1)',
            color: 'rgba(255, 255, 255, 0.5)',
            padding: '4px 10px',
            borderRadius: '6px',
            fontSize: '0.7rem',
            cursor: 'pointer'
          }}
        >
          Copy
        </button>
      </div>
      <pre style={{ margin: 0, padding: '20px', overflowX: 'auto', background: '#1a1b26' }}>
        <code className={`language-${activeLang}`} style={{ fontFamily: "'Fira Code', 'SF Mono', monospace", fontSize: '0.85rem', lineHeight: '1.7', color: '#a9b1d6' }}>
          {activeCode}
        </code>
      </pre>
    </div>
  );
};

  <p>
    The Collections Framework defines root interfaces and standard implementations. Separating core interfaces (such as `List`, `Set`, `Queue`, and `Map`) from their underlying data structures (arrays, linked lists, hash tables, red-black trees) allows for clean polymorphism in application APIs.
  </p>

<section className="content-section">
  <h2 id="list-interface">List Implementations</h2>
  <p>
    Java's two most common `List` implementations have distinct operational characteristics:
  </p>
  <ul>
    <li><strong>ArrayList:</strong> Backed by a dynamically resizable array. Offers $O(1)$ random access reads by index. Array resizing requires copying elements, making middle-list insertions $O(N)$.</li>
    <li><strong>LinkedList:</strong> Backed by a doubly-linked list. Offers $O(1)$ head/tail insertion/deletion, but lacks index-based addressing, making index reads $O(N)$ due to pointer traversal.</li>
  </ul>
</section>

<section className="content-section">
  <h2 id="set-interface">Set Implementations</h2>
  <p>
    `Set` represents unique collections. Key implementations include:
  </p>
  <ul>
    <li><strong>HashSet:</strong> Backed by a `HashMap` instance where values are mapped to a dummy object placeholder. Offers $O(1)$ lookup and insertion. Elements are unordered.</li>
    <li><strong>LinkedHashSet:</strong> Extends `HashSet` but maintains a doubly-linked list running through all of its entries, preserving insertion order.</li>
    <li><strong>TreeSet:</strong> Backed by a Red-Black tree. Keeps elements sorted according to natural ordering or a custom comparator, offering $O(\log N)$ search/insertion operations.</li>
  </ul>
</section>

<section className="content-section">
  <h2 id="map-interface">Map & HashMap Internals</h2>
  <p>
    The internal mechanics of `HashMap` are a classic JVM interview favorite:
  </p>
  <ol>
    <li><strong>Hashing & Buckets:</strong> Keys are passed to a hash function to compute a bucket index (`index = hash(key) & (n - 1)`).</li>
    <li><strong>Hash Collisions:</strong> When multiple keys map to the same bucket index, they are stored in a linked list structure at that slot.</li>
    <li><strong>Treeification:</strong> In Java 8+, if a single bucket's linked list length exceeds the threshold (`TREEIFY_THRESHOLD = 8`) and the map's total capacity is $\ge 64$, the list is converted (treeified) into a Red-Black tree. This changes worst-case lookup from $O(N)$ to $O(\log N)$ to protect against HashDoS security exploits.</li>
  </ol>
</section>

<section className="content-section">
  <h2 id="concurrent-collections">Concurrent & Thread-Safety</h2>
  <p>
    Standard collections are NOT thread-safe. Concurrent modifications can trigger `ConcurrentModificationException` in iterator loops.
  </p>
  <ul>
    <li><strong>Collections.synchronizedMap:</strong> Wraps the map in synchronized blocks. Every read and write locks the entire map, creating severe thread contention bottleneck under high concurrency.</li>
    <li><strong>ConcurrentHashMap:</strong> Employs a lock-stripping architecture. In Java 8+, it uses fine-grained CAS (Compare-And-Swap) operations for empty slots, and locks only the individual bucket node (`synchronized(node)`) for occupied slots. This allows concurrent threads to read and write to different buckets concurrently without blocking.</li>
  </ul>
</section>
